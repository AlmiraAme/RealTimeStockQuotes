  7 /**
  8  * Dit is de constructor van de Timer,
  9  * De Timer kan zo vaak als je wil worden aangemaakt.
 10  * Ik maak er een constructor van om een specifieke object van te maken die zijn eigen methodes en variabels heeft.
 11  * De timer class/object is er om de timer te starten zodat de data steeds wordt geupdate.
 12  *
 13  * De reden dat ik alle functies en variables in de constructor zet is om het zoveel mogenlijk op een class te laten lijken met zijn eigen methodes/variables.
 14  * Om het erin te zetten zien het er ook meer uit als een class.
 15  *
 16  * Used site: http://www.elated.com/articles/javascript-timers-with-settimeout-and-setinterval/
 17  * @constructor : Constructor van de Timer class.
 18  */
 19 function Timer(){
 20     /**
 21      * Het aantal seconde/minuten/uren dat de informatie moet worden geupdate.
 22      * @type {Number} : het aantal seconde/minuten/uren.
 23      */
 24     var delayTime = 5000;
 25 
 26     /**
 27      * De functie start is om de interval te starten,
 28      * De stockprovider moet meegegeven worden om de data steeds te updaten.
 29      * @param stockprovider : de stockprovider dat wordt meegegeven door de constructor waar de yahoo of local wordt opgehaald.
 30      */
 31     this.start = function(stockprovider){
 32         this.interval = setInterval( stockprovider.getData, delayTime);
 33     };
 34 
 35     /**
 36      * Deze functie kan de interval stoppen.
 37      */
 38     this.stop = function(){
 39         clearInterval(this.interval);
 40     };
 41 
 42 }
