  7 /**
  8  * Dit is de constructor van de StockProvider,
  9  * De StockProvider kan zo vaak als je wil worden aangemaakt.
 10  * Ik maak er een constructor van om een specifieke object van te maken die zijn eigen methodes en variabels heeft.
 11  * De stockProvider is een factory pattern.
 12  * De stockprovider geeft een nieuwe Localhelper of YahooAjaxHelper door de functie getData op te roepen.
 13  * In mijn geval set ik al wanneer ik de yahoo wil of de local. Dit doe ik door ze om de 10 keer te wisselen.
 14  * Dus 10 keer geef ik de local dan 10 keer de yahoo en dit steeds om en om.
 15  *
 16  * De reden dat ik alle functies en variables in de constructor zet is om het zoveel mogenlijk op een class te laten lijken met zijn eigen methodes/variables.
 17  * Om het erin te zetten zien het er ook meer uit als een class.
 18  * @constructor : Constructor van de StockProvider class.
 19  */
 20 function StockProvider(){
 21     /**
 22      * Maakt een nieuwe YahooAjaxHelper aan en zet dat in de variable yahoo.
 23      * @type {YahooAjaxHelper} : Geeft een YahooAjaxHelper "class/object".
 24      */
 25     yahoo = new YahooAjaxHelper();
 26 
 27     /**
 28      * Maakt een nieuwe LocalHelper aan en zet dat in de variable local.
 29      * @type {LocalHelper} : Geeft een LocalHelper "class/object".
 30      */
 31     local = new LocalHelper();
 32 
 33     /**
 34      * Dit is om bij te houden het aantal keer dat de methode getdata is aan geroepen,
 35      * zodat er om en om de yahoo en local helpers worden gereturnt
 36      * @type {Number} : count begint met 1.
 37      */
 38     var count = 1;
 39 
 40     /**
 41      * De functie getData is om een nieuwe yahoo of local helper te returnen .
 42      * Na elke keer dat de functie wordt aangeroepen gaat de count omhoog.
 43      * Dan wordt er een if en else gezet, wanneer de getdata 10 keer is aangeroepen (die local mee geeft) geeft hij de yahoo terug.
 44      * Nadat de functie 20 keer is aangeroepen wordt de count gereset en zo geeft de StockProvider factory om en om de local en yahoo en roept de getStockData functie aan.
 45      *
 46      * @param count : het aantal keer dat getdata is opgeroepen.
 47      */
 48     function getData(count) {
 49         /**
 50          * Eerst 10 keer dat data wordt opgeroepen voor de local;
 51          * @type {Number} : 10 keer.
 52          */
 53         var localCount = 10;
 54         /**
 55          * de volgende 10 keer (maximum is dan dus 20 keer) voor de yahoo;
 56          * @type {Number} : 20e keer.
 57          */
 58         var yahooCount = 20;
 59         /**
 60          * reset de count terug naar 1
 61          * @type {Number} :  1 (gereset weer naar het begin)
 62          */
 63         var resetCount = 1;
 64         /**
 65          * hier wordt de functie getdata aangeroepen voor het returnen van de local of yahoo helper.
 66          * @return {*} Het hang van de count af wat hij returnt, of de local.getStockData() of de yahoo.getStockData
 67          */
 68         this.getData = function () {
 69             count++;
 70             if (count <= localCount) {
 71                 return local.getStockData();
 72             }
 73             else {
 74                 if (count === yahooCount) {
 75                     count = resetCount;
 76                 }
 77                 return yahoo.getStockData();
 78             }
 79         };
 80     }
 81 
 82     /**
 83      * Wanneer de functie getData wordt opgeroepen geeft hij de count mee.
 84      */
 85     getData.call(this, count);
 86 }
