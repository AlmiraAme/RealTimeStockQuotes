  7 /**
  8  * Dit is de constructor van de StockView,
  9  * De StockView kan zo vaak als je wil worden aangemaakt.
 10  * Ik maak er een constructor van om een specifieke object van te maken die zijn eigen methodes en variabels heeft.
 11  * De Stockview is er om een tabel te maken met data erin maar ook om de tabel te updaten indien de stocks zijn verandert.
 12  *
 13  * De reden dat ik alle functies en variables in de constructor zet is om het zoveel mogenlijk op een class te laten lijken met zijn eigen methodes/variables.
 14  * Om het erin te zetten zien het er ook meer uit als een class.
 15  * @constructor : Constructor van de StockView class.
 16  */
 17 function StockView(){
 18     /**
 19      * Het Aantal bedrijven (stocks).
 20      * @type {Number} : 24 bedrijven.
 21      */
 22     var numberOfCompanies = 24;
 23 
 24     /**
 25      * het aantal kolommen.
 26      * @type {Number} : 4 is het aantal kolommen dat een rij mag hebben.
 27      */
 28     var numberOfColoms = 4;
 29 
 30     /**
 31      * Hier wordt een stock table gemaakt met data erin.
 32      * Eerst wordt er een tabel gemaakt en een tbody,
 33      * Dat worden er rijen gemaakt door de functie createrows te roepen waarin de cellen worden ingezet in de rijen.
 34      * Nadat de rijen met cellen in de tbody zitten word de tbody in de tabel geplaats en de tabel in de body.
 35      * @param arraylist : arraylijst met de default stocks.
 36      */
 37     this.buildStockTable = function(arraylist){
 38         var table = document.createElement('table');
 39         var tbody = document.createElement('tbody');
 40         this.createRows(tbody,arraylist);
 41         table.appendChild(tbody);
 42         document.body.appendChild(table);
 43     };
 44 
 45     /**
 46      * Dit is de functie createRows dit et aantal rijen creert die er nodig zijn( voor alle bedrijven).
 47      * Eerst wordt eer een loop gemaakt voor het aantal bedrijven waarbij de rijen eerst worden gemaakt.
 48      * Éen rij wordt er extra gemaakt voor de header row.
 49      * Eerst worden de headers gemaakt en in de eerst rij geplaatst.
 50      * Nadat de headers zijn geplaatst worden de overige cellen gecreërt door de functie createCells te roepen.
 51      * Aan de functie geef je elke rij mee met de stock die erin moet worden geset.
 52      * Nadat de cellen zijn gemaakt word elk rij in de tbody gezet.
 53      * @param tbody : De tbody van de tabel
 54      * @param allStocks : de arraylist met de default stocks.
 55      */
 56     this.createRows = function(tbody, allStocks) {
 57         for (var i = 0; i <= numberOfCompanies; i++) {
 58             var row = document.createElement('tr');
 59 
 60             if (i === 0) {
 61                 this.createHeaders(row);
 62             } else {
 63                 row.id = allStocks[i-1].stock[0];
 64                 this.createCells(row, allStocks[i-1].stock);
 65             }
 66             tbody.appendChild(row);
 67         }
 68     };
 69 
 70     /**
 71      * Dit is de functie createHeaders waar de headers worden gemaakt van de tabel.
 72      * Eerst wordt er een loop gemaakt voor elk kolom.
 73      * Voor elk kolom wordt er een header gemaakt.
 74      * het aantal kolommen staan vast en de arraylijst met de headertekst.
 75      * Elke keer dat de header wordt gemaakt wordt de tekst ook gemaakt dan wordt de tekst in de headercel geplaatst en de headercel in de rij.
 76      * @param row : de rij in de tbody/tabel
 77      */
 78     this.createHeaders = function(row) {
 79         /**
 80          * Een arraylijst met de tekst dat in de header moet komen.
 81          * @type {Array} : een array met tekst dat in de header moet komen.
 82          */
 83         var textArray = ['Name','Rate', 'Change +/-', '%+/-'];
 84         for (var j = 0; j < numberOfColoms; j++) {
 85             var tableHeader = document.createElement('th');
 86             var text = document.createTextNode(textArray[j]);
 87             tableHeader.appendChild(text);
 88             row.appendChild(tableHeader);
 89         }
 90     };
 91 
 92     /**
 93      * Dit is de createCells functie waarbij de overige cellen worden gemaakt met de value erin.
 94      * De rij en de stock worden meegegeven.
 95      * Dit weer wordt gezet in de loop en eerst wordt de tablecell gecreeërt en daarna de value.
 96      * Dan wordt de achtergrond kleur geset aan de hand van de change.
 97      * Aan het einde wordt eerst de value/text in de tablecell gezet en de cell in de rij.
 98      *
 99      * @param row : de rij van de tabel/tbody waar de cellen in worden gezet.
100      * @param stock : de stock waar de waardes van in de rij moeten worden gezet.
101      */
102     this.createCells = function(row, stock) {
103         for (var k = 0; k < numberOfColoms; k++) {
104             var tableCell = document.createElement('td');
105             var text = document.createTextNode(stock[k]);
106             this.setBackgroundColor(stock, row);
107             tableCell.appendChild(text);
108             row.appendChild(tableCell);
109         }
110     };
111 
112     /**
113      * Dit is de SetBackgroundColor functie.
114      * De stock en rij wordt meegegeven aan de functie om zo erachter te komen of de achtergrond kleur rood, groen of blauw moet worden.
115      * Eerst wordt er gechecket of er geen verandering is (indien wel wordt de classname verandert naar noChange waarbij de achtergrondkleur blauw wordt)
116      * Dan wordt er gechecket of er wel verandering is en dus groter dan 0 is(indien wel wordt de classname verandert naar Up waarbij de achtergrondkleur groen wordt)
117      * Als laatst wordt er gechecket of er wel verandering is  maar kleiner dan 0 is(indien wel wordt de classname verandert naar Down waarbij de achtergrondkleur Rood wordt)
118      * @param stock : de stock waarbij wordt gekeken of er verandering is in de prijs.
119      * @param row : de rij waar de kleur van wordt verandert door behulp van verandering van classname.
120      */
121     this.setBackgroundColor = function(stock, row) {
122         var changeIndex = 2;
123         if (stock[changeIndex] === 0) {
124             row.className = 'noChange';
125         } else if (stock[changeIndex] > 0.00) {
126             row.className = 'up';
127         } else if (stock[changeIndex] < 0.00) {
128             row.className = 'down';
129         }
130     };
131 
132     /**
133      * De updateTable functie verandert de waarde in de cellen indien die verschillend zijn.
134      * De nieuwe data wordt meegegeven om mee te vergelijken maar ook om de nieuwe data value in de tabel te zetten.
135      * Eerst wordt er een loop gemaakt om door alle stocks heen te gaan en te updaten,
136      * dan wordt er per rij gechecked of de colom value verandert zijn indien dit zo is wordt dit bijgewerkt in de tabel.(dit door de setChangedValues methode te roepen)
137      * @param newdata : de nieuwe data dat in de tabel moet worden gezet(geupdate).
138      */
139     this.updateTable= function(newdata){
140         for(var i = 0; i < numberOfCompanies ; i++){
141             for(var j = 0; j < numberOfColoms ; j++) {
142                 this.setChangedValues(j, newdata[i].stock);
143             }
144         }
145     };
146     /**
147      * Dit is de setChangedValues functie waarbij de nieuwe waarde wordt gechecked naar de waarde in de cell.
148      * indien de waardes anders zijn word de cell waar de waarde anders in is verandert in de nieuwe waarde.
149      * dit door de oude waarde te verwijderen en een nieuwe tekstnode te maken en in de cell te zetten.
150      * ook wordt de achtergrondkleur van de rij geupdate door de functie setBackgroundcolor te roepen.
151      * @param j : de index van de waarde die verandert moet worden indien nodig
152      * @param newstock : de nieuwe stock.
153      */
154     this.setChangedValues = function(j, newstock) {
155         var tablecell = (document.getElementById(newstock[0])).childNodes[j];
156         var oldValue = document.getElementById(newstock[0]).childNodes[j].textContent;
157         if (oldValue !== newstock[j]) {
158             var newValue = document.createTextNode(newstock[j]);
159             this.setBackgroundColor(newstock,(document.getElementById(newstock[0])));
160             this.checkIfHasText(tablecell);
161             tablecell.appendChild(newValue);
162         }
163     };
164 
165     /**
166      * Dit is een extra check om te kijken of er een tekstelement in de cell zit, indien de value niet null is wordt de tekstnode verwijdert uit de cell.
167      * @param tablecell : de cell waar de check moet worden gedaan.
168      */
169     this.checkIfHasText = function(tablecell) {
170         var oldText = tablecell.childNodes[0];
171         if (oldText !== null) {
172             tablecell.removeChild(oldText);
173         }
174     };
